# Senti.AI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a full-stack emotional profiling web app that psychoanalyzes Filipinos based on OPM/international music taste + personality traits, generates AI-powered roasts, and makes sharing viral on social media.

**Architecture:** Next.js 16 App Router with client-side multi-step wizard flow, server-side API routes for Claude AI integration and rate limiting, Framer Motion for animations, html2canvas for share card generation. The app is a single-page wizard (steps 0-7) with additional pages for leaderboard, match, barkada, and history.

**Tech Stack:** Next.js 16, TypeScript, Tailwind CSS v4, Framer Motion, Anthropic Claude API (claude-sonnet-4-20250514), html2canvas, Tone.js

```tsx
// src/components/NeuralNetworkBg.tsx
"use client";

import { useEffect, useRef } from "react";

interface Node {
  x: number;
  y: number;
  vx: number;
  vy: number;
}

export default function NeuralNetworkBg() {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    let animationId: number;
    const nodes: Node[] = [];
    const NODE_COUNT = 40;
    const CONNECTION_DISTANCE = 150;

    function resize() {
      canvas!.width = window.innerWidth;
      canvas!.height = window.innerHeight;
    }
    resize();
    window.addEventListener("resize", resize);

    // Initialize nodes
    for (let i = 0; i < NODE_COUNT; i++) {
      nodes.push({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        vx: (Math.random() - 0.5) * 0.5,
        vy: (Math.random() - 0.5) * 0.5,
      });
    }

    function draw() {
      ctx!.clearRect(0, 0, canvas!.width, canvas!.height);

      // Update and draw nodes
      for (const node of nodes) {
        node.x += node.vx;
        node.y += node.vy;

        // Bounce off edges
        if (node.x < 0 || node.x > canvas!.width) node.vx *= -1;
        if (node.y < 0 || node.y > canvas!.height) node.vy *= -1;

        // Draw node
        ctx!.beginPath();
        ctx!.arc(node.x, node.y, 2, 0, Math.PI * 2);
        ctx!.fillStyle = "rgba(255, 50, 82, 0.3)";
        ctx!.fill();
      }

      // Draw connections
      for (let i = 0; i < nodes.length; i++) {
        for (let j = i + 1; j < nodes.length; j++) {
          const dx = nodes[i].x - nodes[j].x;
          const dy = nodes[i].y - nodes[j].y;
          const dist = Math.sqrt(dx * dx + dy * dy);

          if (dist < CONNECTION_DISTANCE) {
            const opacity = (1 - dist / CONNECTION_DISTANCE) * 0.15;
            ctx!.beginPath();
            ctx!.moveTo(nodes[i].x, nodes[i].y);
            ctx!.lineTo(nodes[j].x, nodes[j].y);
            ctx!.strokeStyle = `rgba(255, 50, 82, ${opacity})`;
            ctx!.lineWidth = 1;
            ctx!.stroke();
          }
        }
      }

      animationId = requestAnimationFrame(draw);
    }

    draw();

    return () => {
      cancelAnimationFrame(animationId);
      window.removeEventListener("resize", resize);
    };
  }, []);

  return (
    <canvas
      ref={canvasRef}
      className="fixed inset-0 pointer-events-none z-0"
      aria-hidden="true"
    />
  );
}
```

```tsx
import { useEffect, useState } from "react";

interface StatBoxProps {
  label: string;
  value: string | number;
  suffix?: string;
  animate?: boolean;
}

export default function StatBox({ label, value, suffix = "", animate = true }: StatBoxProps) {
  const [display, setDisplay] = useState(animate && typeof value === "number" ? 0 : value);

  useEffect(() => {
    if (!animate || typeof value !== "number") return;

    let start = 0;
    const duration = 1500;
    const startTime = Date.now();

    function tick() {
      const elapsed = Date.now() - startTime;
      const progress = Math.min(elapsed / duration, 1);
      // Ease out cubic
      const eased = 1 - Math.pow(1 - progress, 3);
      const current = eased * (value as number);
      setDisplay(Math.round(current * 10) / 10);
      if (progress < 1) requestAnimationFrame(tick);
    }

    requestAnimationFrame(tick);
  }, [value, animate]);

  return (
    <div className="bg-bg-card border border-border-subtle rounded-lg p-4 text-center min-w-[140px]">
      <div className="text-2xl font-bold text-accent font-mono">
        {display}
        {suffix}
      </div>
      <div className="text-xs text-text-muted mt-1 uppercase tracking-wider font-mono">
        {label}
      </div>
    </div>
  );
}
```

```typescript
// src/lib/fingerprint.ts

/** Generate a simple browser fingerprint hash for rate limiting */
export function generateFingerprint(): string {
  const components = [
    navigator.userAgent,
    navigator.language,
    screen.width + "x" + screen.height,
    screen.colorDepth.toString(),
    new Date().getTimezoneOffset().toString(),
    navigator.hardwareConcurrency?.toString() ?? "unknown",
  ];

  // Simple hash
  const str = components.join("|");
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    const char = str.charCodeAt(i);
    hash = (hash << 5) - hash + char;
    hash |= 0;
  }
  return Math.abs(hash).toString(36);
}

/** Get remaining analyses count from localStorage */
export function getRemainingAnalyses(fingerprint: string): number {
  const key = `senti_usage_${fingerprint}`;
  const today = new Date().toDateString();

  try {
    const stored = localStorage.getItem(key);
    if (stored) {
      const data = JSON.parse(stored);
      if (data.date === today) {
        return Math.max(0, 2 - data.count);
      }
    }
  } catch {}
  return 2;
}

/** Record an analysis usage */
export function recordAnalysis(fingerprint: string): void {
  const key = `senti_usage_${fingerprint}`;
  const today = new Date().toDateString();

  try {
    const stored = localStorage.getItem(key);
    let count = 1;
    if (stored) {
      const data = JSON.parse(stored);
      if (data.date === today) {
        count = data.count + 1;
      }
    }
    localStorage.setItem(key, JSON.stringify({ date: today, count }));
  } catch {}
}
```

```tsx
  onSongsChange: (songs: Song[]) => void;
  onNext: () => void;
}

export default function SongInputStep({ songs, onSongsChange, onNext }: SongInputStepProps) {
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<Song[]>([]);
  const [showDropdown, setShowDropdown] = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);
  const dropdownRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (query.length >= 1) {
      setResults(searchSongs(query).slice(0, 8));
      setShowDropdown(true);
    } else {
      setResults([]);
      setShowDropdown(false);
    }
  }, [query]);

  // Close dropdown on click outside
  useEffect(() => {
    function handleClick(e: MouseEvent) {
      if (
        dropdownRef.current &&
        !dropdownRef.current.contains(e.target as Node) &&
        inputRef.current &&
        !inputRef.current.contains(e.target as Node)
      ) {
        setShowDropdown(false);
      }
    }
    document.addEventListener("mousedown", handleClick);
    return () => document.removeEventListener("mousedown", handleClick);
  }, []);

  function addSong(song: Song) {
    if (songs.length >= 5) return;
    if (songs.some((s) => s.title === song.title && s.artist === song.artist)) return;
    onSongsChange([...songs, song]);
    setQuery("");
    setShowDropdown(false);
    inputRef.current?.focus();
  }

  function removeSong(index: number) {
    onSongsChange(songs.filter((_, i) => i !== index));
  }

  function handleKeyDown(e: React.KeyboardEvent) {
    if (e.key === "Enter" && query.trim() && results.length > 0) {
      addSong(results[0]);
    } else if (e.key === "Enter" && query.trim() && results.length === 0) {
      // Add custom song
      const customSong: Song = {
        title: query.trim(),
        artist: "Unknown",
        mood: "unknown",
        painIndex: 5.5,
      };
      addSong(customSong);
    }
  }

  return (
    <div className="space-y-6">
      <StepIndicator current={1} />

      <div className="space-y-2">
        <h2 className="text-2xl font-bold">Your Hugot Playlist</h2>
        <p className="text-text-secondary text-sm">
          Select 3-5 songs that define your emotional state rn
        </p>
      </div>

      {/* Search input */}
      <div className="relative">
        <input
          ref={inputRef}
          type="text"
          value={query}
          onChange={(e) => setQuery(e.target.value)}
          onKeyDown={handleKeyDown}
          placeholder="Search songs or artists..."
          className="w-full bg-bg-card border border-border-subtle rounded-lg px-4 py-3 text-text-primary placeholder:text-text-muted focus:outline-none focus:border-accent/50 font-mono text-sm"
        />

        {/* Dropdown */}
        {showDropdown && results.length > 0 && (
          <div
            ref={dropdownRef}
            className="absolute top-full left-0 right-0 mt-1 bg-[#111118] border border-border-subtle rounded-lg overflow-hidden z-20 max-h-[300px] overflow-y-auto"
          >
            {results.map((song, i) => (
              <button
                key={`${song.title}-${song.artist}`}
                onClick={() => addSong(song)}
                className="w-full flex items-center justify-between px-4 py-3 hover:bg-accent/10 transition-colors text-left cursor-pointer"
              >
                <div>
                  <span className="text-accent text-sm">{song.title}</span>
                  <span className="text-text-muted text-sm ml-2">— {song.artist}</span>
                </div>
                <span className="text-text-muted text-xs font-mono">{song.painIndex}</span>
              </button>
            ))}
          </div>
        )}
      </div>

      {/* Selected songs */}
      <div className="space-y-2">
        <p className="text-xs text-text-muted font-mono">{songs.length}/5 songs added</p>

        onClick={onNext}
        disabled={songs.length < 3}

        Next
```

```typescript
  const entry = ipStore.get(ip);

  if (!entry || now > entry.resetAt) {
    ipStore.set(ip, { count: 1, resetAt: now + WINDOW_MS });
    return { allowed: true, remaining: MAX_REQUESTS - 1 };
  }

  if (entry.count >= MAX_REQUESTS) {
    return { allowed: false, remaining: 0 };
  }

  entry.count++;
  return { allowed: true, remaining: MAX_REQUESTS - entry.count };
}
```

import { NextRequest } from "next/server";
import Anthropic from "@anthropic-ai/sdk";
import { buildPrompt } from "@/lib/buildPrompt";
import { checkRateLimit } from "@/lib/rateLimit";
import { AnalysisRequest, ProfileResult } from "@/lib/types";

const anthropic = new Anthropic();

export async function POST(request: NextRequest) {
  // Get client IP
  const ip =
    request.headers.get("x-forwarded-for")?.split(",")[0]?.trim() ||
    request.headers.get("x-real-ip") ||
    "unknown";

  // Rate limit check
  const { allowed, remaining } = checkRateLimit(ip);
  if (!allowed) {
    return Response.json(
      {
        error: "rate_limited",
        message:
          "The creator of Senti.AI believed in second chances... not a third though. 'D ako bobo.",
      },
      { status: 429 }
    );
  }

  // Parse and validate request
  let body: AnalysisRequest;
  try {
    body = await request.json();
  } catch {
    return Response.json({ error: "Invalid request body" }, { status: 400 });
  }

  if (
    !body.songs?.length ||
    !body.mbti ||
    !body.attachmentStyle ||
    !body.loveLanguage ||
    !body.zodiac
  ) {
    return Response.json({ error: "Missing required fields" }, { status: 400 });
  }

  // Build prompt and call Claude
  const { system, user } = buildPrompt(
    body.songs,
    body.mbti,
    body.attachmentStyle,
    body.loveLanguage,
    body.zodiac
  );

  try {
    const message = await anthropic.messages.create({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1500,
      system,
      messages: [{ role: "user", content: user }],
    });

    // Extract text from response
    const textBlock = message.content.find((b) => b.type === "text");
    if (!textBlock || textBlock.type !== "text") {
      throw new Error("No text in response");
    }

    // Parse JSON from response (strip any markdown code fences if present)
    let jsonStr = textBlock.text.trim();
    if (jsonStr.startsWith("```")) {
      jsonStr = jsonStr.replace(/^```(?:json)?\n?/, "").replace(/\n?```$/, "");
    }

    const result: ProfileResult = JSON.parse(jsonStr);

    return Response.json({ result, remaining });
  } catch (error) {
    console.error("Claude API error:", error);
    return Response.json(
      { error: "analysis_failed", message: "API call failed" },
      { status: 500 }
    );
  }
}

```typescript
import html2canvas from "html2canvas";

export async function captureCard(element: HTMLElement): Promise<Blob> {
  const canvas = await html2canvas(element, {
    backgroundColor: "#0a0a0f",
    scale: 1, // Already at target resolution
    useCORS: true,
    logging: false,
  });

  return new Promise((resolve, reject) => {
    canvas.toBlob(
      (blob) => {
        if (blob) resolve(blob);
        else reject(new Error("Failed to generate image"));
      },
      "image/png",
      1.0
    );
  });
}

export async function shareOrDownload(blob: Blob, filename: string): Promise<void> {
  const file = new File([blob], filename, { type: "image/png" });

  // Try Web Share API (mobile)
  if (navigator.share && navigator.canShare?.({ files: [file] })) {
    try {
      await navigator.share({
        files: [file],
        title: "My Senti.AI Results",
        text: "Check out my emotional damage assessment 😭",
      });
      return;
    } catch (err) {
      // User cancelled or share failed, fall through to download
      if ((err as Error).name === "AbortError") return;
    }
  }

  // Fallback: download
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}
```

```typescript
const SPOTIFY_AUTH_URL = "https://accounts.spotify.com/authorize";

const SCOPES = "user-top-read";

function generateCodeVerifier(): string {
  const array = new Uint8Array(64);
  crypto.getRandomValues(array);
  return btoa(String.fromCharCode(...array))
    .replace(/\+/g, "-")
    .replace(/\//g, "_")
    .replace(/=+$/, "");
}

async function generateCodeChallenge(verifier: string): Promise<string> {
  const encoder = new TextEncoder();
  const data = encoder.encode(verifier);
  const digest = await crypto.subtle.digest("SHA-256", data);
  return btoa(String.fromCharCode(...new Uint8Array(digest)))
    .replace(/\+/g, "-")
    .replace(/\//g, "_")
    .replace(/=+$/, "");
}

export async function initiateSpotifyAuth(): Promise<void> {
  const clientId = process.env.NEXT_PUBLIC_SPOTIFY_CLIENT_ID;
  const redirectUri = process.env.NEXT_PUBLIC_SPOTIFY_REDIRECT_URI || `${window.location.origin}/callback`;

  if (!clientId) {
    console.error("Missing SPOTIFY_CLIENT_ID");
    return;
  }

  const verifier = generateCodeVerifier();
  const challenge = await generateCodeChallenge(verifier);

  sessionStorage.setItem("spotify_code_verifier", verifier);

  const params = new URLSearchParams({
    client_id: clientId,
    response_type: "code",
    redirect_uri: redirectUri,
    scope: SCOPES,
    code_challenge_method: "S256",
    code_challenge: challenge,
  });

  window.location.href = `${SPOTIFY_AUTH_URL}?${params}`;
}

export async function exchangeCode(code: string): Promise<string | null> {
  const clientId = process.env.NEXT_PUBLIC_SPOTIFY_CLIENT_ID;
  const redirectUri = process.env.NEXT_PUBLIC_SPOTIFY_REDIRECT_URI || `${window.location.origin}/callback`;
  const verifier = sessionStorage.getItem("spotify_code_verifier");

  if (!clientId || !verifier) return null;

  try {
    const res = await fetch(SPOTIFY_TOKEN_URL, {
      method: "POST",
      headers: { "Content-Type": "application/x-www-form-urlencoded" },
      body: new URLSearchParams({
        client_id: clientId,
        grant_type: "authorization_code",
        code,
        redirect_uri: redirectUri,
        code_verifier: verifier,
      }),
    });

    if (!res.ok) return null;
    const data = await res.json();
    return data.access_token;
  } catch {
    return null;
  }
}

export async function fetchTopTracks(token: string): Promise<SpotifyTrack[]> {
  try {
    // Try short term first
    let res = await fetch(
      "https://api.spotify.com/v1/me/top/tracks?time_range=short_term&limit=30",
      { headers: { Authorization: `Bearer ${token}` } }
    );

    if (!res.ok) throw new Error("Failed to fetch");
    let data = await res.json();

    // Fall back to medium term if short term is sparse
    if (data.items.length < 5) {
      res = await fetch(
        "https://api.spotify.com/v1/me/top/tracks?time_range=medium_term&limit=30",
        { headers: { Authorization: `Bearer ${token}` } }
      );
      if (res.ok) data = await res.json();
    }

    return data.items || [];
  } catch {
    return [];
  }
}
```

// src/app/api/match/route.ts
import { NextRequest } from "next/server";
import Anthropic from "@anthropic-ai/sdk";
import { buildMatchPrompt } from "@/lib/buildMatchPrompt";
import { checkRateLimit } from "@/lib/rateLimit";
import { UserProfile, MatchResult } from "@/lib/types";

const anthropic = new Anthropic();

// In-memory match store (use KV in production)
const matchStore = new Map<string, { profileA: UserProfile; profileB?: UserProfile; matchResult?: MatchResult; createdAt: number }>();

// Clean up old entries periodically
function cleanup() {
  const cutoff = Date.now() - 48 * 60 * 60 * 1000; // 48 hours
  for (const [key, val] of matchStore) {
    if (val.createdAt < cutoff) matchStore.delete(key);
  }
}

export async function POST(request: NextRequest) {
  cleanup();
  const body = await request.json();

  // Create match
  if (body.action === "create") {
    const id = Math.random().toString(36).slice(2, 10);
    matchStore.set(id, {
      profileA: body.profile,
      createdAt: Date.now(),
    });
    return Response.json({ id });
  }

  // Complete match
  if (body.action === "complete") {
    const match = matchStore.get(body.id);
    if (!match) {
      return Response.json({ error: "Match not found or expired" }, { status: 404 });
    }

    // Rate limit User B
    const ip = request.headers.get("x-forwarded-for")?.split(",")[0]?.trim() || "unknown";
    const { allowed } = checkRateLimit(ip);
    if (!allowed) {
      return Response.json({ error: "rate_limited" }, { status: 429 });
    }

    match.profileB = body.profile;

    // Generate match analysis
    const { system, user } = buildMatchPrompt(match.profileA, match.profileB);

    try {
      const message = await anthropic.messages.create({
        model: "claude-sonnet-4-20250514",
        max_tokens: 1200,
        system,
        messages: [{ role: "user", content: user }],
      });

      const textBlock = message.content.find((b) => b.type === "text");
      if (!textBlock || textBlock.type !== "text") throw new Error("No text");

      let jsonStr = textBlock.text.trim();
      if (jsonStr.startsWith("```")) {
        jsonStr = jsonStr.replace(/^```(?:json)?\n?/, "").replace(/\n?```$/, "");
      }

      const matchResult: MatchResult = JSON.parse(jsonStr);
      match.matchResult = matchResult;

      return Response.json({
        matchResult,
        profileA: match.profileA,
        profileB: match.profileB,
      });
    } catch (err) {
      console.error("Match API error:", err);
      return Response.json({ error: "analysis_failed" }, { status: 500 });
    }
  }

  return Response.json({ error: "Invalid action" }, { status: 400 });
}

export async function GET(request: NextRequest) {
  const url = new URL(request.url);
  const id = url.searchParams.get("id");
  if (!id) return Response.json({ error: "Missing id" }, { status: 400 });

  const match = matchStore.get(id);
  if (!match) return Response.json({ error: "Not found" }, { status: 404 });

  return Response.json({
    profileA: match.profileA,
    profileB: match.profileB,
    matchResult: match.matchResult,
    hasProfileB: !!match.profileB,
  });
}

```typescript
// src/app/api/barkada/route.ts
import { NextRequest } from "next/server";
import { UserProfile } from "@/lib/types";

interface BarkadaMember {
  nickname: string;
  profile: UserProfile;
}

interface BarkadaGroup {
  members: BarkadaMember[];
  createdAt: number;
}

const groupStore = new Map<string, BarkadaGroup>();

function cleanup() {
  const cutoff = Date.now() - 7 * 24 * 60 * 60 * 1000; // 7 days
  for (const [key, val] of groupStore) {
    if (val.createdAt < cutoff) groupStore.delete(key);
  }
}

export async function POST(request: NextRequest) {
  cleanup();
  const body = await request.json();

  if (body.action === "create") {
    const id = Math.random().toString(36).slice(2, 10);
    groupStore.set(id, {
      members: [{ nickname: body.nickname, profile: body.profile }],
      createdAt: Date.now(),
    });
    return Response.json({ id });
  }

  if (body.action === "join") {
    const group = groupStore.get(body.id);
    if (!group) return Response.json({ error: "Group not found" }, { status: 404 });
    if (group.members.length >= 10) return Response.json({ error: "Group is full" }, { status: 400 });

    group.members.push({ nickname: body.nickname, profile: body.profile });
    return Response.json({ success: true, members: group.members });
  }

  return Response.json({ error: "Invalid action" }, { status: 400 });
}

export async function GET(request: NextRequest) {
  const url = new URL(request.url);
  const id = url.searchParams.get("id");
  if (!id) return Response.json({ error: "Missing id" }, { status: 400 });

  const group = groupStore.get(id);
  if (!group) return Response.json({ error: "Not found" }, { status: 404 });

  return Response.json({ members: group.members });
}
```

```typescript
    .sort((a, b) => b.score - a.score)
    .slice(0, 50);

export async function POST(request: NextRequest) {
  const body = await request.json();

  const entry: LeaderboardEntry = {
    score: body.score,
    mbti: body.mbti,
    attachmentStyle: body.attachmentStyle,
    zodiac: body.zodiac,
    threat_level: body.threat_level,
    timestamp: Date.now(),
  };

  leaderboard.push(entry);
  leaderboard.sort((a, b) => b.score - a.score);
  if (leaderboard.length > 100) leaderboard.length = 100;

  const rank = leaderboard.findIndex(
    (e) => e.score === entry.score && e.timestamp === entry.timestamp
  ) + 1;

  return Response.json({ rank, total: leaderboard.length });
```

```tsx
              <div className="bg-bg-card border border-border-subtle rounded-lg p-4">
                <div className="text-xs text-text-muted font-mono mb-4">DAMAGE OVER TIME</div>
                <div className="flex items-end gap-2 h-[120px]">
                  {entries.map((e, i) => {
                    const height = (e.score / 10) * 100;
                    const color = THREAT_COLORS[e.threat_level];
                    return (
                      <div key={i} className="flex-1 flex flex-col items-center gap-1">
                        <span className="text-xs font-mono" style={{ color }}>
                          {e.score}
                        </span>
                        <div
                          className="w-full rounded-t"
                          style={{
                            height: `${height}%`,
                            backgroundColor: color,
                            opacity: 0.7,
                          }}
                        />
                        <span className="text-[10px] text-text-muted">
                          {new Date(e.timestamp).toLocaleDateString("en-PH", {
                            month: "short",
                            day: "numeric",
                          })}
                        </span>
                      </div>
                    );
                  })}
                </div>
              </div>

              {/* Score change */}
              {entries.length >= 2 && (
                <div className="bg-bg-card border border-border-subtle rounded-lg p-4">
                  <div className="text-sm text-text-primary">
                    {(() => {
                      const latest = entries[entries.length - 1].score;
                      const prev = entries[entries.length - 2].score;
                      const diff = latest - prev;
                      const pct = Math.abs(Math.round((diff / prev) * 100));
                      if (diff > 0) return `Your emotional damage has increased ${pct}% since last time. Seek help.`;
                      if (diff < 0) return `Your emotional damage decreased ${pct}%. Suspicious. Are you lying?`;
                      return `Same score as last time. Consistent emotional damage. At least you're stable.`;
                    })()}
                  </div>
                </div>
              )}

              {/* Entry list */}
              <div className="space-y-2">
                {[...entries].reverse().map((e, i) => (
                  <div key={i} className="bg-bg-card border border-border-subtle rounded-lg p-3">
                    <div className="flex justify-between items-center">
                      <span className="text-sm text-text-primary truncate flex-1 mr-2">
                        {e.headline}
                      </span>
                      <span
                        className="font-mono font-bold text-sm"
                        style={{ color: THREAT_COLORS[e.threat_level] }}
                      >
                        {e.score}
                      </span>
                    </div>
                    <div className="text-xs text-text-muted mt-1">
                      {e.mbti} · {new Date(e.timestamp).toLocaleDateString()}
                    </div>
                  </div>
                ))}
              </div>
            </>
          )}

          <div className="text-center">
            <a href="/" className="text-accent font-mono text-sm hover:underline">
              ← Take another scan
            </a>
          </div>
        </div>
      </main>
    </div>
  );
}
```

- [ ] **Step 1: Create useSound hook**

```typescript
// src/hooks/useSound.ts
"use client";

import { useCallback, useRef, useState } from "react";

export function useSound() {
  const [muted, setMuted] = useState(true);
  const audioCtxRef = useRef<AudioContext | null>(null);

  function getCtx() {
    if (!audioCtxRef.current) {
      audioCtxRef.current = new AudioContext();
    }
    return audioCtxRef.current;
  }

  const playClick = useCallback(() => {
    if (muted) return;
    const ctx = getCtx();
    const osc = ctx.createOscillator();
    const gain = ctx.createGain();
    osc.connect(gain);
    gain.connect(ctx.destination);
    osc.frequency.setValueAtTime(800, ctx.currentTime);
    osc.frequency.exponentialRampToValueAtTime(400, ctx.currentTime + 0.05);
    gain.gain.setValueAtTime(0.1, ctx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.05);
    osc.start(ctx.currentTime);
    osc.stop(ctx.currentTime + 0.05);
  }, [muted]);

  const playLockIn = useCallback(() => {
    if (muted) return;
    const ctx = getCtx();
    const osc = ctx.createOscillator();
    const gain = ctx.createGain();
    osc.connect(gain);
    gain.connect(ctx.destination);
    osc.frequency.setValueAtTime(400, ctx.currentTime);
    osc.frequency.exponentialRampToValueAtTime(800, ctx.currentTime + 0.15);
    gain.gain.setValueAtTime(0.15, ctx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.2);
    osc.start(ctx.currentTime);
    osc.stop(ctx.currentTime + 0.2);
  }, [muted]);

  const playAlert = useCallback(() => {
    if (muted) return;
    const ctx = getCtx();
    // Two-tone alert
    [0, 0.15].forEach((offset) => {
      const osc = ctx.createOscillator();
      const gain = ctx.createGain();
      osc.connect(gain);
      gain.connect(ctx.destination);
      osc.type = "square";
      osc.frequency.setValueAtTime(offset === 0 ? 440 : 660, ctx.currentTime + offset);
      gain.gain.setValueAtTime(0.08, ctx.currentTime + offset);
      gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + offset + 0.12);
      osc.start(ctx.currentTime + offset);
      osc.stop(ctx.currentTime + offset + 0.12);
    });
  }, [muted]);

  return { muted, setMuted, playClick, playLockIn, playAlert };
}
```
